Don't Forget The Deps

useEffect(() => {
  if (props?.displayError) {
    form.validateForm();
  }
}, []);

consider this code sample

  if (props?.displayError) {
  }

the expectation here is to validate the form everytime displayError changes to true. Do you think this expectation is met?

  if (props?.displayError) {
  }

useEffect executes every time its dependencies change

  if (props?.displayError) {
  }

as there are no dependencies here it will only execute once and that's it

  if (props?.displayError) {
  }

i.e., even if the displayError comes back to true the form is not revalidated

  if (props?.displayError) {
  }

to fix we should add props.displayError to the dependency array

  if (props?.displayError) {
  }

now what should happen when the form changes and displayError is still true?

  if (props?.displayError) {
  }
}, [form, props?.displayError]);

expectation is, it should revalidate the form. This does not happen!!

  }
}, [form, props?.displayError]);

  }

#1 Custom Hook

const confirmationDialogStateMachine = {
  none: {
    on: {
      OPEN_SAVE_CONFIRMATION: 'saveConfirmation',
      OPEN_DELETE_CONFIRMATION: 'deleteConfirmation',
    }
  },
  saveConfirmation: {
    on: {
      CLOSE: 'none',
    }
  },
  deleteConfirmation: {
    on: {
      CLOSE: 'none',
    }
  }
}
​
function confirmationDialogReducer(state, { type }) {
  return confirmationDialogStateMachine[state].on[type]
}
​
function MyEditForm() {
  const [visibleDialogType, send /* or dispatch */] = useReducer(confirmationDialogReducer, 'none')
​
  const transition = (dialogType) => send({ type: dialogType });
​
  return (
    <>
      <form>{/* ... */}</form>
​
      <button onClick={() => transition('OPEN_SAVE_CONFIRMATION')}>Save</button>
      <button onClick={() => transition('OPEN_DELETE_CONFIRMATION')}>Delete</button>
​
      <ConfirmationDialog
        message="Are you sure you want to save?"
        onClose={() => transition('CLOSE')}
        open={visibleDialogType === 'saveConfirmation'}
      />
​
      <ConfirmationDialog
        message="Are you sure you want to delete?"
        onClose={() => transition('CLOSE')}
        open={visibleDialogType === 'deleteConfirmation'}
      />
    </>
  )
}
